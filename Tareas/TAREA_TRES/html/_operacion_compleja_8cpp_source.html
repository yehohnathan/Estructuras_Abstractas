48template <typename T>
49void OperacionCompleja<T>::conversorMatriz(const Matriz<T>& matriz){
50 /* Se crea un ciclo for anidado para convertir los datos en el vector real pero
51 distribuido como si fuera un número complejo a un verdadero vector complejo */
52
53 /* Se multiplcan los iteradores * 2 para ajustar los espacios */
54 for (int i = 0; i < matriz.getFilas()*2; i++) {
55 for (int j = 0; j < matriz.getColumnas()*2; j +=2) {
56 /* Se obtiene el valor de la parte real y la parte imaginaria*/
57 complex<double> numeroComplejo(matriz.get(i, j), matriz.get(i, (j + 1)));
58
59 if(matriz1.empty()){
60 matrizCompleja1.push_back(numeroComplejo);
61 }
62 else if(matriz2.empty()){
63 matrizCompleja2.push_back(numeroComplejo);
64 }
65 }
66 i ++;
67 }
68
69 /* La explicación de por qué este ciclo for funciona es por lo siguiente:
70 1. El vector matriz dice que es de tamaño NxM, pero en realidad es de tamaño Nx(M*2);
71 ocultando el hecho de que realmente no es una matriz compleja, unicamente es real.
72
73 2. En el caso de tener una matriz:
74 1+2j 3+4j
75 5+6j 7+8j
76 Eso es igual a matriz = [1, 2, 3, 4, 5, 6, 7, 8]
77
78 3. matriz.get(i,j) = i*matriz.getColumnas()+ j. Por lo que en el caso de tener una matriz 2x2:
79 Para el elemento 1x1:
80 real = get(0,0) = 0*2+0 = matriz[0]
81 imaginario = get(0,1) = 0*2+1 = matriz[1]
82
83 Para el elemento 1x2:
84 real = get(0,2) = 0*2+2 = matriz[2]
85 imaginario = get(0,3) = 0*2+3 = matriz[3]
86
87 Para el elemento 2x1:
88 No se puede hacer:
89 real = get(1,0) = 1*2+0 = matriz[2]
90 Por lo que se le suma i + 1 al final del ciclo for de j:
91 real = get(2,0) = 2*2+0 = matriz[4]
92 imaginario = get(2,1) = 2*2+1 = matriz[5]
93
94 Para el elemento 2x2:
95 real = get(2,2) = 2*2+2 = matriz[6]
96 imaginario = get(2,3) = 2*2+3 = matriz[7]
97 */
98}
